import random def shuffleArray(array): length = len(array) swaps = length*3 lastIndex = length-1 for i in range(swaps): i = random.randint(0,lastIndex) j = random.randint(0,lastIndex) temp = array[i] array[i] = array[j] array[j] = temp return array
